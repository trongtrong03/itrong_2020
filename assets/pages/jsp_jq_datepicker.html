Plugin 來源：flatpickr

Plugin 所需文件：
jquery.js
flatpickr.js
flatpickr.css


將以下程式碼內容複製至 .html 文件內：

<input type="text" class="datepicker" placeholder="請選擇日期時間"/>

引用 JS 文件：

<script src="https://cdn.jsdelivr.net/npm/flatpickr"></script>

引用 CSS 文件：

<link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/flatpickr/dist/flatpickr.min.css">


將以下程式碼內容複製至 .js 文件內或 <script> 元素中：

$(function() {
    var doSomething = function() {
        console.log(fp[0].currentYear);
    }
    var options = {
        enableTime: true,
        dateFormat: "Y-m-d H:i",
        altInput: true,
        altFormat: "F j, Y H:i",
        minuteIncrement: 0,
        //maxDate: "today",
        //minDate: new Date(2018, 0, 1),
        //plugins: [new rangePlugin({ input: #input2"})]
        onChange: doSomething
    };

    const fp = $(".datepicker").flatpickr(options);
    var date = new Date();
    fp[0].setDate(date);
    console.log(fp[0].currentYear);
});

maxDate 指的是日曆範圍的可選最大日期。
minDate 指的是日曆範圍的可選最小日期。
minuteIncrement 指的是分鐘的增加幅度，0 為無法選擇分鐘。
