Plugin 所需文件：
jquery.js


將以下程式碼內容複製至 .html 文件內：

<div class="news-wrap">
    <ul class="news-item">
        <li><a href="#0">標題A</a></li>
        <li><a href="#0">標題B</a></li>
        <li><a href="#0">標題C</a></li>
    </ul>
</div>


將以下程式碼內容複製至 .js 文件內或 <script> 元素中：

var news = null;
function newsTicker() {
  $(".news-item")
    .filter(function(item) {
        return !$(this).is(news);
    })
    .each(function() {
        $(this)
        .find("li:first")
        .slideUp(function() {
            var newsitem = $(this).closest(".news-item");
            $(this)
                .appendTo(newsitem)
                .slideDown();
        });
    });
}
setInterval(newsTicker, 5000);

$(function() {  
    $(".news-item").hover(
        function() {
            news = $(this);
        },
        function() {
            news = null;
        }
    ); // set timeout
});


僅列出此效果必要的 CSS，請將以下程式碼內容複製至 .css 文件內或 <style> 元素中：

.news-list {
    height: 6rem;
    overflow: hidden;
}
